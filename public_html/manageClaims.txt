    <script>
        // Dummy data for demonstration
        const existingClaims = [
            { user: 'John Doe', amount: 'R10000', status: 'Pending' },
            { user: 'Jane Smith', amount: 'R15000', status: 'Approved' },
            { user: 'Bob Johnson', amount: 'R5000', status: 'Denied' }
        ];

        // Function to populate existing claims
        function populateClaims() {
            const claimsList = document.getElementById('claims-list');
            existingClaims.forEach(claim => {
                const claimElement = document.createElement('div');
                claimElement.className = 'claim';
                claimElement.innerHTML = `
                    <h3>${claim.user}</h3>
                    <p>Amount: ${claim.amount}</p>
                    <p>Status: ${claim.status}</p>`;
                claimsList.appendChild(claimElement);
            });
        }

        // Function to populate users claims table
        function populateUsersClaims() {
            const usersClaimsList = document.getElementById('users-claims-list');
            usersClaimsList.innerHTML = ''; // Clear existing data
            existingClaims.forEach(claim => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${claim.user}</td>
                    <td>${claim.amount}</td>
                    <td>${claim.status}</td>`;
                usersClaimsList.appendChild(row);
            });
        }

        // Function to filter users claims
        function filterUsersClaims() {
            const searchTerm = document.getElementById('search-input').value.toLowerCase();
            const filterType = document.getElementById('filter-select').value;
            const filteredClaims = existingClaims.filter(claim => {
                const matchesSearchTerm = claim.user.toLowerCase().includes(searchTerm);
                const matchesFilterType = filterType === 'all' || claim.status === filterType;
                return matchesSearchTerm && matchesFilterType;
            });

            const usersClaimsList = document.getElementById('users-claims-list');
            usersClaimsList.innerHTML = ''; // Clear existing data
            filteredClaims.forEach(claim => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${claim.user}</td>
                    <td>${claim.amount}</td>
                    <td>${claim.status}</td>`;
                usersClaimsList.appendChild(row);
            });
        }

        document.addEventListener('DOMContentLoaded', function() {
            populateClaims();
            populateUsersClaims();

            document.getElementById('search-input').addEventListener('input', filterUsersClaims);
            document.getElementById('filter-select').addEventListener('change', filterUsersClaims);
        });

        document.getElementById('add-claim-form').addEventListener('submit', function(event) {
            event.preventDefault();
            const claimName = document.getElementById('claim-name').value;
            const claimAmount = document.getElementById('claim-amount').value;
            const claimStatus = document.getElementById('claim-status').value;
            const claimElement = document.createElement('div');
            claimElement.className = 'claim';
            claimElement.innerHTML = `
                <h3>${claimName}</h3>
                <p>Amount: R${claimAmount}</p>
                <p>Status: ${claimStatus}</p>`;
            document.getElementById('claims-list').appendChild(claimElement);
            alert('New claim added');
        });
    </script>
